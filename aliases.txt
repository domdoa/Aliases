function gb() {
  git branch
}

function gcb() {
  git checkout -b "$1"
}

function gcm() {
  git checkout master
}

function gcmp() {
  git checkout master
  git pull
}

function gp() {
  git pull
}

function gc(){
  git checkout "$1"
}

function gbd(){
  git branch -D $1
}

function gs(){
  git stash
}

function gsp(){
  git stash pop
}

function gsl(){
  git stash list
}
